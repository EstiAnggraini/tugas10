🏭 Pengantar Factory Constructors

Slide 2
  • Apa itu? Constructor khusus yang tidak selalu membuat instance baru.
  • Keyword: factory
  • Perbedaan dengan generative constructor:
    • Generative: Selalu membuat instance baru.
    • Factory: Bisa mengembalikan instance yang sudah ada, atau instance subclass.

🔧 Sintaks Factory Constructor

// Nama: Esti Anggraini
// Nim: 251410031
// Kelas: SI2A                
class Example {
  // Private constructor agar tidak bisa dibuat dari luar
  Example._();

  // Factory constructor yang memanggil constructor private
  factory Example.create() => Example._();

  // Static method yang juga memanggil constructor private
  static Example createStatic() => Example._();
}

void main() {
  // Penggunaan factory constructor
  var e1 = Example.create();        // Factory constructor
  
  // Penggunaan static method
  var e2 = Example.createStatic();  // Static method
  
  // var e3 = Example(); // Error: constructor private (tidak bisa diakses)
  
  print('e1 dibuat melalui factory constructor: $e1');
  print('e2 dibuat melalui static method: $e2');
  print('Apakah e1 dan e2 instance yang sama? ${identical(e1, e2)}'); // false
  
  // Demonstrasi bahwa setiap pemanggilan membuat instance baru
  var e3 = Example.create();
  var e4 = Example.createStatic();
  
  print('e1 dan e3 berbeda? ${!identical(e1, e3)}'); // true - instance berbeda
  print('e2 dan e4 berbeda? ${!identical(e2, e4)}'); // true - instance berbeda
  print('e3 dan e4 berbeda? ${!identical(e3, e4)}'); // true - instance berbeda
  
  // Atau bisa juga pakai e4 untuk sesuatu yang bermakna
  print('Total instances yang dibuat: 4 (e1, e2, e3, e4)');
}

// Nama: Esti Anggraini
// Nim: 251410031
// Kelas: SI2A                  
  class ConnectionPool {
  final String host;
  final int port;
  // Cache berdasarkan parameter
  static final Map _cache = {};
  
  // Private constructor
  ConnectionPool._(this.host, this.port);
  
  // Factory dengan caching
  factory ConnectionPool(String host, int port) {
    var key = '$host:$port';
    // Return dari cache jika ada
    if (_cache.containsKey(key)) {
      print('Returning cached connection for $key');
      return _cache[key]!;
    }
    // Buat baru dan cache
    print('Creating new connection for $key');
    var connection = ConnectionPool._(host, port);
    _cache[key] = connection;
    return connection;
  }
  
  void query(String sql) {
    print('Executing: "$sql" on $host:$port');
  }
}

void main() {
  print('--- Creating conn1 (localhost:5432) ---');
  var conn1 = ConnectionPool('localhost', 5432);
  conn1.query('SELECT * FROM users;');
  print('');
  
  print('--- Creating conn2 (localhost:5432 - should be cached) ---');
  var conn2 = ConnectionPool('localhost', 5432); // Dari cache
  conn2.query('INSERT INTO products (name) VALUES ("Laptop");');
  print('');
  
  print('--- Creating conn3 (db.example.com:3306 - new connection) ---');
  var conn3 = ConnectionPool('db.example.com', 3306); // Baru
  conn3.query('UPDATE orders SET status="shipped" WHERE id=101;');
  print('');
  
  // Demonstrate that conn1 and conn2 are the same instance
  print('Are conn1 and conn2 the same instance? ${identical(conn1, conn2)}'); // true
  
  // Demonstrate that conn1 and conn3 are different instances
  print('Are conn1 and conn3 the same instance? ${identical(conn1, conn3)}'); // false
}
✨ Output demonstrasi: koneksi yang sama untuk host:port yang sama, mengurangi overhead.

class User {
  final String id;
  final String name;
  final String email;
  
  User._({required this.id, required this.name, required this.email});
  
  factory User.fromJson(Map json) {
    return User._(
      id: json['id']?.toString() ?? '',
      name: json['name']?.toString() ?? '',
      email: json['email']?.toString() ?? '',
    );
  }
  
  factory User.guest() {
    return User._(
      id: 'guest_${DateTime.now().millisecondsSinceEpoch}',
      name: 'Guest',
      email: 'guest@example.com',
    );
  }
  
  factory User.create(String name, String email) {
    if (name.isEmpty) throw ArgumentError('Name cannot be empty');
    if (!email.contains('@')) throw ArgumentError('Invalid email');
    return User._(
      id: 'user_${DateTime.now().millisecondsSinceEpoch}',
      name: name,
      email: email,
    );
  }
  
  @override
  String toString() => 'User(id: $id, name: $name, email: $email)';
}

void main() {
  final jsonUser = User.fromJson({'id': '123', 'name': 'Alice', 'email': 'alice@example.com'});
  final guest = User.guest();
  final newUser = User.create('Bob', 'bob@example.com');
  
  print('JSON User: $jsonUser');
  print('Guest User: $guest');
  print('New User: $newUser');
}

abstract class Response {
  final T? data;
  final String? error;
  
  Response({this.data, this.error});
  
  // Factory untuk success
  factory Response.success(T data) = SuccessResponse;
  
  // Factory untuk error
  factory Response.error(String error) = ErrorResponse;
}

class SuccessResponse extends Response {
  SuccessResponse(T data) : super(data: data);
}

class ErrorResponse extends Response {
  ErrorResponse(String error) : super(error: error);
}

void main() {
  // Dart bisa menginfer tipe data secara otomatis
  final successResponse = Response.success(100);  // Otomatis Response
  final errorResponse = Response.error("Something went wrong");  // Otomatis Response
  
  print('Success response data: ${successResponse.data}');
  print('Error response error: ${errorResponse.error}');
  
  // Type checking
  if (successResponse is SuccessResponse) {
    print('This is a SuccessResponse with int data');
  }
  
  if (errorResponse is ErrorResponse) {
    print('This is an ErrorResponse with String error');
  }
  
  // Menampilkan tipe data
  print('\nRuntime types:');
  print('successResponse type: ${successResponse.runtimeType}');
  print('errorResponse type: ${errorResponse.runtimeType}');
  
  // Contoh tambahan
  print('\n--- Additional Examples ---');
  
  final stringSuccess = Response.success("Hello World");
  print('String success: ${stringSuccess.data}');
  
  final boolSuccess = Response.success(true);
  print('Bool success: ${boolSuccess.data}');
  
  final doubleError = Response.error("Invalid number format");
  print('Double error: ${doubleError.error}');
}

🎨 Factory untuk Immutable Objects dengan Cache

Slide 12
// Nama: Esti Anggraini
// Nim: 251410031
// Kelas: SI2A               
  class Color {
  final int red;
  final int green;
  final int blue;

  static final Map _cache = {};

  // Private constructor
  Color._(this.red, this.green, this.blue);

  // Factory dengan cache
  factory Color.rgb(int red, int green, int blue) {
    if (red < 0 || red > 255 || green < 0 || green > 255 || blue < 0 || blue > 255) {
      throw ArgumentError('RGB values must be 0-255');
    }
    var key = '$red,$green,$blue';
    if (!_cache.containsKey(key)) {
      _cache[key] = Color._(red, green, blue);
    }
    return _cache[key]!;
  }

  // Named factories untuk warna umum
  factory Color.red() => Color.rgb(255, 0, 0);
  factory Color.green() => Color.rgb(0, 255, 0);
  factory Color.blue() => Color.rgb(0, 0, 255);
  factory Color.white() => Color.rgb(255, 255, 255);
  factory Color.black() => Color.rgb(0, 0, 0);

  @override
  String toString() => 'Color($red, $green, $blue)';
}

void main() {
  var red1 = Color.red();
  var red2 = Color.rgb(255, 0, 0);
  print(identical(red1, red2)); // true - instance sama
  print(red1);
  print(red2);
}

✅ Ringkasan & Best Practices

Slide 15-16

🎯 Poin Penting Factory Constructor

  • Factory constructor: factory class.name() { ... }
  • Bisa mengembalikan: instance yang ada (cache), subclass, atau instance baru.
  • Use cases: singleton, caching, object pooling, subclass selection, immutable objects pooling.

📌 Best Practices:

  • Gunakan factory untuk: Object pooling/caching, Singleton pattern, Complex object creation, Subclass selection.
  • Gunakan generative constructor untuk: Simple object creation, initialization lists, required parameters.
  • Validasi: Lakukan validasi di factory sebelum membuat objek (contoh pada Color).
⚡ Factory constructor sangat berguna untuk mengendalikan pembuatan objek, menghemat memori, dan menyediakan API yang bersih.

// ==================== ANIMAL FACTORY ====================
class AnimalFactory {
  static final AnimalFactory _instance = AnimalFactory._internal();
  final Map _animalCache = {};
  final List _validAnimalTypes = [
    'lion', 'elephant', 'tiger', 'giraffe', 
    'eagle', 'penguin', 'snake', 'crocodile', 
    'dolphin', 'shark'
  ];
  
  int _cacheHits = 0;
  int _cacheMisses = 0;
  int _totalAnimalsCreated = 0;
  
  factory AnimalFactory() => _instance;
  AnimalFactory._internal();
  
  String _validateAnimalType(String type) {
    final normalizedType = type.toLowerCase().trim();
    
    if (normalizedType.isEmpty) {
      throw ArgumentError('❌ Animal type cannot be empty!');
    }
    
    if (!_validAnimalTypes.contains(normalizedType)) {
      throw ArgumentError(
        '❌ Invalid animal type: "$type"\n'
        '   Supported types: ${_validAnimalTypes.join(", ")}'
      );
    }
    
    return normalizedType;
  }
  
  Animal createAnimal(String type) {
    print('\n🔍 Creating animal: "$type"');
    
    final validatedType = _validateAnimalType(type);
    
    // Check cache
    if (_animalCache.containsKey(validatedType)) {
      _cacheHits++;
      print('✅ CACHE HIT! Returning cached ${validatedType.toUpperCase()}');
      print('📊 Cache stats - Hits: $_cacheHits, Misses: $_cacheMisses');
      return _animalCache[validatedType]!;
    }
    
    // Create new animal
    _cacheMisses++;
    _totalAnimalsCreated++;
    Animal animal;
    
    switch (validatedType) {
      case 'lion': animal = Lion(); break;
      case 'elephant': animal = Elephant(); break;
      case 'tiger': animal = Tiger(); break;
      case 'giraffe': animal = Giraffe(); break;
      case 'eagle': animal = Eagle(); break;
      case 'penguin': animal = Penguin(); break;
      case 'snake': animal = Snake(); break;
      case 'crocodile': animal = Crocodile(); break;
      case 'dolphin': animal = Dolphin(); break;
      case 'shark': animal = Shark(); break;
      default: throw ArgumentError('❌ Unsupported animal type: $validatedType');
    }
    
    _animalCache[validatedType] = animal;
    print('🆕 CACHE MISS! Created new ${validatedType.toUpperCase()}');
    print('📊 Cache stats - Hits: $_cacheHits, Misses: $_cacheMisses');
    
    return animal;
  }
  
  void printCacheStats() {
    print('\n╔════════════════════════════════════════════╗');
    print('║        ANIMAL FACTORY STATISTICS          ║');
    print('╠════════════════════════════════════════════╣');
    print('║ Total animals created: ${_totalAnimalsCreated.toString().padLeft(12)} ║');
    print('║ Cache size:            ${_animalCache.length.toString().padLeft(12)} ║');
    print('║ Cache hits:            ${_cacheHits.toString().padLeft(12)} ║');
    print('║ Cache misses:          ${_cacheMisses.toString().padLeft(12)} ║');
    print('║ Hit ratio:             ${(_cacheHits / (_cacheHits + _cacheMisses) * 100).toStringAsFixed(1).padLeft(12)}% ║');
    print('╚════════════════════════════════════════════╝\n');
  }
  
  void clearCache() {
    _animalCache.clear();
    _cacheHits = 0;
    _cacheMisses = 0;
    print('\n🗑️  Animal cache cleared!');
  }
  
  List getSupportedAnimals() => List.unmodifiable(_validAnimalTypes);
}
